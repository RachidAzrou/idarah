script.js

// Timeline: logo tonen -> logotekst verdwijnt -> kaart schuift uit witte streep -> kop verschijnt
const stage = document.getElementById('stage');

// start sequentie
requestAnimationFrame(() => {
  stage.classList.add('step-1');        // verberg logotekst
  setTimeout(() => {
    stage.classList.add('step-2');      // kaart verschijnt vanaf divider
    setTimeout(() => {
      stage.classList.add('step-3');    // kop + subkop verschijnen
    }, 700);
  }, 1100);
});

// wachtwoord toggle
const btn = document.querySelector('.peek');
btn.addEventListener('click', () => {
  const input = document.getElementById('pwd');
  const isPw = input.getAttribute('type') === 'password';
  input.setAttribute('type', isPw ? 'text' : 'password');
  btn.setAttribute('aria-label', isPw ? 'Verberg wachtwoord' : 'Toon wachtwoord');
});
